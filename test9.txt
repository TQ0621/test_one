#include<stdio.h>
#pragma warning(disable:4996)

void sort(int* a, int len)
{
	
	for (int j = len - 1; j > 1; j--)
	{
		int *maxid = a;
		for (int i = 1; i <= j; i++)
		{
			if (*(maxid) < *(a + i))
			{
				maxid = (a + i);
			}
		}
		int tmp = *(a + j);
		*(a + j) = *maxid;
		*maxid = tmp;
	}
}
int main()
{
	int i=0,a[10] = { 0 };
	printf("请输入10个要排序数。\n");
	int len = sizeof(a) / sizeof(a[0]);
	for (i = 0; i < len; i++)
	{
		scanf("%d", &a[i]);
	}
	sort(a, len);
	for (i = 0; i < len; i++)
	{
		printf("%d ", a[i]);
	}
	return 0;
}
